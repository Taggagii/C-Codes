\documentclass[tikz,border=10pt]{standalone}
\usetikzlibrary{calc,patterns,angles,quotes}
\usepackage{xfp}

\begin{document}
	\begin{tikzpicture}[scale=0.5pt] 
		\newcommand\qoffset{11} 
		\newcommand\smallx{8}
		\newcommand\smally{4} 
		\newcommand\py{\fpeval{\smally * (\qoffset / (\qoffset - \smallx))}}
		\coordinate (origin) at (0, 0);
		\coordinate (ycap) at ($(origin) + (0, 20)$);
		\coordinate (xcap) at ($(origin) + (20, 0)$);		
		\coordinate (smallorigin) at ($(origin) + (\smallx, \smally)$);
		\coordinate (smallycap) at ($(ycap) + (\smallx, 0)$);
		\coordinate (smallxcap) at ($(xcap) + (0, \smally)$);
		\coordinate (q) at ($(origin) + (\qoffset, 0)$);	
		\coordinate (p) at ($(origin) + (0, \py)$);	

		\draw (origin) rectangle ($(origin) + (1, 1)$);
		\draw (origin) -- (ycap) node [black, midway, left]{$y$};
		\draw (origin) -- (xcap) node [black, midway, below]{$x$};
		\draw (smallorigin) -- (smallycap); 
		\draw (smallorigin) -- (smallxcap);
		\draw[dashed, gray] ($(origin) + (8, 0)$) -- (smallorigin) node [black, midway, left] {4};
		\draw[dashed, gray] ($(origin) + (0, 4)$) -- (smallorigin) node [black, midway, below] {8};
		\fill (q) circle(0.1) node[anchor = north]{$q$};
		\fill (p) circle(0.1) node[anchor = east]{$p$};
		\draw (p) -- (q) node[midway, above right] {$L$};
		\pic [draw, -, "$\theta$", angle eccentricity = 1.5] {angle = p--q--origin};
	\end{tikzpicture}
\end{document}
